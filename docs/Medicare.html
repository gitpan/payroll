Medicare->calculate(
    gross,  # gross pay
    date,    # date of payment
)
{
  iterate over sorted keys descending in data hash.  
    Compare $date to keys.  
    Stop when $date >= key.  
    set $foundDate = $key.
    If date < all keys, that is an error.  

  return data{$foundDate}->{rate} * gross
}
